```latexsvg
\documentclass{standalone}
\usepackage{tikz}
\usepackage{graphicx}

\usetikzlibrary{calc,angles}

\begin{document}

\begin{tikzpicture}
    % 定义长方形的顶点（宽度8，高度4，方便计算）
    \coordinate (A) at (0,0);   % 左下角
    \coordinate (B) at (8,0);  % 右下角
    \coordinate (C) at (8,4);  % 右上角
    \coordinate (D) at (0,4);  % 左上角

    % 定义直角三角形的顶点
    \coordinate (E) at (8,2);   % 直角顶点在右边上 (8,2)
    \coordinate (G) at (7.5,4); % 锐角顶点在上边上 (7.5,4)
    % 绘制长方形
    \draw (A) -- (B) -- (C) -- (D)  -- (A) -- (E) -- (G) -- cycle;

    % 标记顶点
    \node[below left] at (A) {A};
    \node[below right] at (B) {B};
    \node[above right] at (C) {C};
    \node[above left] at (D) {D};
    \node[right] at (E) {E};
    \node[above] at (G) {G};

   
     % 标记角 GAB 为 alpha（确保标记锐角）
    \pic [
        draw, 
        angle radius=1cm, 
        angle eccentricity=1.3, 
        pic text=$\alpha$, 
        pic text options={color=black, font=\small} % 显式设置标签属性
    ] {angle = E--A--G};

    % 标记角 EAB 为 beta（确保标记锐角）
    \pic [
        draw, 
        angle radius=0.8cm, 
        angle eccentricity=1.3, 
        pic text=$\beta$, 
        pic text options={color=black, font=\small} % 显式设置标签属性
    ] {angle = B--A--E};
    % 可选：添加坐标辅助线
    \draw[dashed, gray] (E) -- (8,0); % 右边到底边的垂线
    \draw[dashed, gray] (G) -- (7.5,0); % 上边顶点到底边的垂线
\end{tikzpicture}

\end{document}
```

```latexsvg
\documentclass[tikz,border=3mm]{standalone}
\usepackage{tikz}
\usetikzlibrary{calc, angles}

\begin{document}
\begin{tikzpicture}[scale=0.8]
    % 定义正方形尺寸
    \def\a{4}   % 边长

    % 绘制正方形并命名关键点
    \draw[thick] (0,0) coordinate (A) rectangle (\a,\a);
    
    % 定义三角形顶点
    \coordinate (C) at (3,\a);    % 上边点
    \coordinate (B) at (\a,2);    % 右边直角顶点

    % 绘制红色直角三角形
    \draw[red, thick] (A) -- (B) node[right] {$B(4,2)$} 
                          -- (C) node[above] {$C(3,4)$} 
                          -- cycle;

    % 添加过 C 点的垂线（新增代码）
    \draw[dashed, gray] (C) -- (3,0) node[midway, right] {Vertical Line}; % 从 C(3,4) 垂直到底边 (3,0)

    % 使用专业角度标记
    \pic[draw=red, angle radius=2mm, angle eccentricity=0.7] {right angle = A--B--C};

    % 添加辅助线说明
    \draw[dashed, blue!50] (A) -- (C) node[midway, sloped, above] {Diameter};
    \draw[dashed, blue] ($(A)!0.5!(C)$) circle (2.5cm);

    % 坐标标注
    \node[below left] at (A) {$A(0,0)$};
    \node[above right] at (\a,\a) {$D(4,4)$};
    \draw[stealth-stealth, gray] (0,\a+0.5) node[above] {Top} -- (0,0) -- (\a+0.5,0) node[right] {Right};
\end{tikzpicture}
\end{document}
```
